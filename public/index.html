async function fetchGame() { const name = document.getElementById('name').value.trim(); const tag = document.getElementById('tag').value.trim(); if(!name || !tag) return alert("닉네임과 태그를 입력하세요."); document.getElementById('loader').classList.remove('hidden'); document.getElementById('dashboard').classList.add('hidden'); try { const res = await fetch('/analyze', { method: 'POST', headers: {'Content-Type': 'application/json'}, body: JSON.stringify({ name, tag }) }); // 서버 응답이 왔을 때만 처리 const data = await res.json(); if(data.error) { alert(data.error); document.getElementById('loader').classList.add('hidden'); return; } // 데이터 렌더링 (닉네임 포함) renderTeams(data.pList, data.version); fullStrat = data.strategy; document.getElementById('strat-title').innerText = `⚡ ${data.myChamp} 실전 훈수`; showSummary(); document.getElementById('loader').classList.add('hidden'); document.getElementById('dashboard').classList.remove('hidden'); } catch (err) { console.error(err); alert("분석 중 오류가 발생했습니다. 다시 시도해 주세요."); document.getElementById('loader').classList.add('hidden'); } }
