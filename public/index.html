// index.html의 analyze 함수 부분을 아래 내용으로 교체하거나 전체를 덮어쓰세요. async function analyze() { const nameInput = document.getElementById('name').value.trim(); let tagInput = document.getElementById('tag').value.trim(); if(!nameInput || !tagInput) return alert('닉네임과 태그를 모두 입력해주세요!'); // 🔥 핵심: #이 포함되어 있다면 #을 제거하고 그 뒤의 텍스트만 추출 const cleanTag = tagInput.startsWith('#') ? tagInput.substring(1) : tagInput; document.getElementById('loading').style.display = 'block'; document.getElementById('board').innerHTML = ''; document.getElementById('strategy').style.display = 'none'; try { const res = await fetch('/analyze', { method: 'POST', headers: {'Content-Type': 'application/json'}, body: JSON.stringify({ summonerName: nameInput, tagLine: cleanTag // 깨끗해진 태그 전달 }) }); const data = await res.json(); if(data.error) throw new Error(data.error); render(data.teams); document.getElementById('strategy').innerText = data.strategy; document.getElementById('strategy').style.display = 'block'; } catch(e) { alert("에러: " + e.message); } document.getElementById('loading').style.display = 'none'; }
