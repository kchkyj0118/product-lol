async function fetchGame() { const n = document.getElementById('name').value.trim(); const t = document.getElementById('tag').value.trim(); if(!n || !t) return alert("닉네임#태그를 입력하세요."); // 화면 초기화 및 로딩 표시 document.getElementById('loader').classList.remove('hidden'); document.getElementById('dashboard').classList.add('hidden'); try { const res = await fetch('/analyze', { method: 'POST', headers: {'Content-Type': 'application/json'}, body: JSON.stringify({ name: n, tag: t }) }); const data = await res.json(); // 백엔드에서 보낸 커스텀 에러 처리 if(data.error) { alert(data.error); document.getElementById('loader').classList.add('hidden'); return; } // 닉네임이 포함된 리스트 렌더링 renderTeams(data.pList, data.ver); fullStrat = data.strategy; document.getElementById('strat-title').innerText = `⚡ ${data.myChamp} 실전 훈수`; showSummary(); document.getElementById('loader').classList.add('hidden'); document.getElementById('dashboard').classList.remove('hidden'); } catch (err) { alert("연결 오류가 발생했습니다."); document.getElementById('loader').classList.add('hidden'); } }
